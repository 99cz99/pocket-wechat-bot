5. 创建 AI 桥接脚本
5编写 claude-fast.js — stdin→API→stdout⏱ 3 分钟

这是整个机器人的核心——从 stdin 读用户消息 → 调 DeepSeek API → 回复写到 stdout。

将下面完整代码保存为文件，只改一处：API_KEY 换成你的 DeepSeek Key。

javascript#!/data/data/com.termux/files/usr/bin/node
// claude-fast — cc-connect AI 桥接脚本
const https = require('https');
const http = require('http');
const fs = require('fs');
const path = require('path');
const readline = require('readline');

const API_KEY = 'sk-你的DeepSeekKey';   // ← 改这里
const BASE_URL = 'https://api.deepseek.com/v1';
const MODEL = 'deepseek-v4-pro';
const WORK_DIR = process.env.HOME + '/cc-connect';

let systemPrompt = '';
try {
  systemPrompt = fs.readFileSync(path.join(WORK_DIR, 'CLAUDE.md'), 'utf-8');
} catch (e) {
  systemPrompt = '用简体中文回复，简洁直接。';
}

const conversation = [{ role: 'system', content: systemPrompt }];

function emit(obj) { process.stdout.write(JSON.stringify(obj) + '\n'); }

function extractUserText(line) {
  let obj;
  try { obj = JSON.parse(line); } catch (_) { return line.trim(); }
  if (obj.type === 'user' && obj.message?.content) {
    const c = obj.message.content;
    if (typeof c === 'string') return c.trim();
    if (Array.isArray(c)) return c.filter(x => x.type === 'text').map(x => x.text).join('\n').trim();
  }
  if (obj.type === 'system') return '';
  return '';
}

function callAPI(userMsg, callback) {
  if (!userMsg) { callback(null, ''); return; }
  const url = new URL(BASE_URL + '/chat/completions');
  const transport = url.protocol === 'https:' ? https : http;
  conversation.push({ role: 'user', content: userMsg });
  const body = JSON.stringify({
    model: MODEL, messages: conversation,
    stream: false, max_tokens: 2048, temperature: 0.8
  });
  const req = transport.request({
    hostname: url.hostname, port: url.port || (url.protocol === 'https:' ? 443 : 80),
    path: url.pathname, method: 'POST',
    headers: {
      'Content-Type': 'application/json',
      'Authorization': 'Bearer ' + API_KEY,
      'Content-Length': Buffer.byteLength(body)
    },
    timeout: 120000
  }, (res) => {
    let data = '';
    res.on('data', chunk => data += chunk);
    res.on('end', () => {
      try {
        const json = JSON.parse(data);
        const text = json.choices?.[0]?.message?.content || '';
        if (text) { conversation.push({ role: 'assistant', content: text }); }
        else { conversation.pop(); }
        while (conversation.length > 29) conversation.splice(1, 2);
        callback(null, text);
      } catch (e) { conversation.pop(); callback(null, ''); }
    });
  });
  req.on('error', (e) => { conversation.pop(); callback(e); });
  req.on('timeout', () => { req.destroy(); callback(new Error('timeout')); });
  req.write(body); req.end();
}

const rl = readline.createInterface({ input: process.stdin });
rl.on('line', (line) => {
  const msg = extractUserText(line.trim());
  if (!msg) return;
  callAPI(msg, (err, text) => {
    if (err) {
      emit({ type: 'assistant', message: { content: [{ type: 'text', text: '错误: ' + err.message }] } });
    } else if (text) {
      emit({ type: 'assistant', message: { content: [{ type: 'text', text: text }] } });
    }
    emit({ type: 'result', subtype: 'success' });
  });
});

9. 创建启动脚本
9编写 start-nene.sh — 一键启动⏱ 1 分钟

启动脚本负责：检查旧实例 → 清理锁文件 → 用 proot 隔离启动 cc-connect。

bash#!/data/data/com.termux/files/usr/bin/bash
echo ""
echo "  =============================="
echo "    nene - cc-connect 微信机器人"
echo "  =============================="
echo ""

CONFIG="$HOME/.cc-connect/config.toml"
LOCK="$HOME/.cc-connect/.config.toml.lock"

if [ -f "$LOCK" ]; then
    echo "[!] 已有实例在运行，先停止..."
    proot \
      -b /data/data/com.termux/files/usr:/usr \
      -b $HOME:/home \
      -b /dev:/dev -b /proc:/proc \
      /usr/bin/env PATH=/usr/bin:/usr/local/bin:/home/bin \
      $HOME/bin/cc-connect --config "$CONFIG" --force 2>/dev/null
    rm -f "$LOCK"
    sleep 1
fi

echo "[*] 启动中..."

SSL_CERT_FILE=/data/data/com.termux/files/usr/etc/tls/cert.pem \
proot \
  -b /data/data/com.termux/files/usr:/usr \
  -b $HOME:/home \
  -b /dev:/dev -b /proc:/proc \
  /usr/bin/env PATH=/usr/bin:/usr/local/bin:/home/bin \
  $HOME/bin/cc-connect --config "$CONFIG" &

sleep 3
echo "[*] 管理面板: http://127.0.0.1:9820"
echo "[*] 已启动~"

🔍
proot 是什么？Android 的 libc 是 Bionic，而 cc-connect（Go 编译的）需要 glibc 环境。proot 创建 Linux 命名空间隔离，让 cc-connect 以为自己在标准 Linux 上运行。

📖 使用方法
1 基本对话

用另一个微信号给机器人的微信发任意消息，bot 会自动回复。首次回复 10-30 秒，后续 3-10 秒。

bot 有对话记忆——同一段会话内记得之前聊过什么。如果超过 idle_timeout_mins（默认 120 分钟）没说话，会话重置。

2 获取你的微信 ID（/whoami）

给 bot 发：

/whoami

bot 会返回你的微信 ID（类似 oXXXXxxx@im.wechat）。拿到后填进 config.toml 的 admin_from，你就是管理员了。

3 管理面板

手机浏览器打开：

http://127.0.0.1:9820

可以查看 bot 运行状态、活跃会话、消息日志。token 是 config.toml 里 management.token 的值。

4 修改人设（实时生效）

编辑 ~/cc-connect/CLAUDE.md，改完保存。无需重启，下一轮对话自动用新人设。

这意味着你可以随时调整 bot 的性格、回复风格、规则，改完发条消息就能看到效果。

5 多轮对话 & 会话管理

bot 会记住同一段会话内的上下文。聊到一半切话题也能接上。

会话在以下情况重置：

  • 超过 idle_timeout_mins 分钟没说话（默认 120）
  • bot 重启
  • 对话历史超过长度上限自动裁剪旧消息
6 添加更多微信好友使用

默认 allow_from = "*"，任何微信好友都能和 bot 对话，且每人有独立会话。

如果想限定只有特定人能聊，把 allow_from 改成白名单（逗号分隔的 wxid 列表）。

❓ 常见问题
Q消息发出去没回复？
  1. bot 在跑吗？pgrep -f cc-connect
  2. API Key 对了吗？cat ~/bin/claude-fast.js | grep API_KEY
  3. 微信 token 过期了吗？需要重新扫码
Q回复很慢？

首条 5-30 秒正常。后续 3-10 秒。持续慢：精简 CLAUDE.md、检查网络。

Q怎么换 AI？

改 ~/bin/claude-fast.js 里的 API_KEY、BASE_URL、MODEL。兼容 OpenAI 格式的 API 都可以。

⚠️ 局限与注意事项
1 Android 杀后台 — 可能需要小窗运行 Termux

即使开了 termux-wake-lock、关了电池优化、锁了多任务卡片，部分手机厂商（小米/华为/OPPO/vivo）仍会在切后台几分钟后杀死 Termux 进程。这是 Android 厂商的省电策略，无法彻底解决。

应对：把 Termux 开在分屏/小窗模式，让它始终在前台可见。bot 就能一直跑。缺点是占屏幕一角。

2 微信 token 会过期

微信扫码获取的 token 不是永久的，可能几天或几周后失效。表现为消息发出去没有回复，日志里可能看到平台断开。

应对：重新执行 ~/bin/cc-connect weixin setup --project nene 扫码，然后把新的 token 和 account_id 更新到 config.toml，重启 bot。

3 微信机器人存在封号风险

这是微信非官方使用方式，不是正规 API。微信有权封禁账号。绝对不要用主号。

降低风险：控制消息频率、不拉群、不加好友、不在群聊里高频回复。

4 ADB 方式不能完全替代手机操作

ADB 的 run-as 环境缺少 Termux 的完整 PATH，proot、bash 等命令可能找不到。遇到 command not found 时，必须在手机 Termux 里手动执行对应命令。

而且每次重启手机后 USB 调试可能自动关闭，需要重新连电脑授权。

5 DeepSeek API 有时会挂

DeepSeek 服务偶尔不稳定（超时、500 错误、模型繁忙）。表现为消息发出去很久没回，或者回复「解析失败」。

应对：等几分钟重试。或者换一个 API 供应商（改 claude-fast.js 里的 BASE_URL 和 API_KEY）。

6 大文件 CLAUDE.md 拉低首次响应速度

如果 CLAUDE.md 很大（如 45KB），首次消息需等待 10-30 秒（加载 system prompt + API 首响应）。后续消息因为上下文复用会快很多（3-10 秒），但如果对话太长导致历史被裁剪，下次又会慢。

应对：精简 CLAUDE.md 到 5-10KB，只保留核心规则。人格细节可以写在外部文件让 AI 自行读取。

7 手机性能影响体验

老旧手机运行 Node.js + proot 会吃力。CPU 满载时回复可能 >30 秒，或 Termux 被系统以「资源占用过高」为由杀死。

建议：用闲置的中高端手机（骁龙 7 系以上，4GB+ RAM），不要用日常主力机。

8 cc-connect 项目本身的风险

cc-connect 是第三方开源项目，GitHub 仓库可能停更、被删、或被微信协议变更导致不可用。没有官方 SLA，坏了只能自己修。
